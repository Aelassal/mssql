Q1: Sessions for a date — summary with verification
-----------------------------------------------------
SELECT
    ca.SessionID,
    ca.SessionDate,
    ca.EmployeeID,
    e.EmployeeName AS CashierName,
    ca.InvoiceCount,
    ca.SalesInvoiceCount,
    ca.ReturnInvoiceCount,
    ca.LineCount AS StoredLineCount,
    COUNT(id.InvoiceDetailID) AS CalculatedLineCount,
    ca.ItemsCount AS StoredItemsCount,
    CAST(SUM(id.Quantity) AS INT) AS CalculatedItemsCount,
    ROUND(ca.NetTotal, 2) AS NetTotal,
    ROUND(ca.ActualAmount, 2) AS ActualAmount,
    ROUND(ca.NetTotalDiff, 2) AS NetTotalDiff,
    ca.CashierClosed,
    ca.SessionClosed,
    CASE WHEN ca.LineCount = COUNT(id.InvoiceDetailID) THEN 'OK' ELSE 'MISMATCH' END AS LineCountMatch,
    CASE WHEN ca.ItemsCount = CAST(SUM(id.Quantity) AS INT) THEN 'OK' ELSE 'MISMATCH' END AS ItemsCountMatch
FROM tblCashierActivity ca
LEFT JOIN tblEmployeesInfo e ON ca.EmployeeID = e.EmployeeID
INNER JOIN tblInvoice i ON ca.SessionID = i.SessionID
INNER JOIN tblInvoiceDetail id ON i.InvoiceID = id.InvoiceID
WHERE ca.SessionDate >= '2026-04-07'
  AND ca.SessionDate < '2026-04-08'
GROUP BY
    ca.SessionID, ca.SessionDate, ca.EmployeeID, e.EmployeeName,
    ca.InvoiceCount, ca.SalesInvoiceCount, ca.ReturnInvoiceCount,
    ca.LineCount, ca.ItemsCount, ca.NetTotal, ca.ActualAmount, ca.NetTotalDiff,
    ca.CashierClosed, ca.SessionClosed
ORDER BY ca.SessionID DESC;

Q4: Products sold in a session (non-returned only) with totals
---------------------------------------------------------------
SELECT
    ROW_NUMBER() OVER (ORDER BY SUM(id.Quantity) DESC) AS Rank,
    id.ItemID, id.ItemName, id.EnglishName, id.BarCode, id.UnitName,
    SUM(id.Quantity) AS TotalQuantity,
    COUNT(*) AS TimesSold,
    ROUND(MIN(id.UnitPrice), 2) AS MinPrice,
    ROUND(MAX(id.UnitPrice), 2) AS MaxPrice,
    ROUND(AVG(id.UnitPrice), 2) AS AvgPrice,
    ROUND(SUM(id.SubTotal), 2) AS SubTotal,
    ROUND(SUM(id.LineDiscount), 2) AS TotalDiscount
FROM tblInvoice i
INNER JOIN tblInvoiceDetail id ON i.InvoiceID = id.InvoiceID
WHERE i.SessionID = 12390 AND i.IsReturned = 0
GROUP BY id.ItemID, id.ItemName, id.EnglishName, id.BarCode, id.UnitName

UNION ALL

SELECT
    9999, NULL, 'TOTALS', NULL, NULL, NULL,
    SUM(id.Quantity), COUNT(*), NULL, NULL, NULL,
    ROUND(SUM(id.SubTotal), 2),
    ROUND(SUM(id.LineDiscount), 2)
FROM tblInvoice i
INNER JOIN tblInvoiceDetail id ON i.InvoiceID = id.InvoiceID
WHERE i.SessionID = 12390 AND i.IsReturned = 0
ORDER BY Rank;


Q5: How NetTotal is calculated — sales minus returns
------------------------------------------------------
SELECT
    'Sales (IsReturned=0)' AS Step,
    ROUND(SUM(i.NetTotal), 2) AS Amount
FROM tblInvoice i
WHERE i.SessionID = 12390 AND i.IsReturned = 0

UNION ALL

SELECT
    'Minus CRA Returns' AS Step,
    -ROUND(ISNULL(SUM(r.ReturnAmount), 0), 2)
FROM tblCashierActivityReturnAmount r
WHERE r.SessionID = 12390

UNION ALL

SELECT
    'Calculated Total' AS Step,
    ROUND(
        (SELECT ISNULL(SUM(NetTotal), 0) FROM tblInvoice WHERE SessionID = 12390 AND IsReturned = 0) -
        (SELECT ISNULL(SUM(ReturnAmount), 0) FROM tblCashierActivityReturnAmount WHERE SessionID = 12390)
    , 2)

UNION ALL

SELECT
    'Stored in Session' AS Step,
    ROUND(ca.NetTotal, 2)
FROM tblCashierActivity ca
WHERE ca.SessionID = 12390;

Q6: Return vouchers for a session (CRA + ReturnCode + ZATCA details)
---------------------------------------------------------------------
SELECT
    cra.ReturnCode, cra.SessionID,
    ROUND(cra.ReturnAmount, 2) AS ReturnAmount,
    cra.ReturnDate,
    rc.ReturnInvoiceID AS OriginalInvoiceID,
    rc.InvoiceID AS ReturnReceiptID,
    rc.Used, rc.Canceled,
    zd.ItemID, zd.ItemName, zd.Quantity, zd.UnitPrice,
    ROUND(zd.SubTotal, 2) AS SubTotal, zd.TaxAmount, zd.TaxPercent
FROM tblCashierActivityReturnAmount cra
JOIN tblInvoiceReturnCode rc ON cra.ReturnCode = rc.ReturnCode
LEFT JOIN tblZatcaCreditNote zcn ON rc.InvoiceID = zcn.InvoiceID
LEFT JOIN tblZatcaCreditNoteDetail zd ON zcn.InvoiceID = zd.InvoiceID
WHERE cra.SessionID = 12390
ORDER BY cra.ReturnCode, zd.LineNumber;

Q10b: All credit notes issued on a date (from tblZatcaCreditNote)
-------------------------------------------------------------------
NOTE: Uses 02:00-02:00 business day window.
This may show MORE credit notes than Q10 because some belong to
sessions with NULL SessionDate or sessions that started on a prior date.
The code only processes returns via CRA vouchers linked to sessions for
the sync date — credit notes from orphan sessions will be missed.

SELECT
    zcn.InvoiceID AS CreditNoteID,
    zcn.ReturnInvoiceID AS OriginalInvoiceID,
    zcn.SessionID,
    ca.SessionDate,
    zcn.CustomerName,
    ROUND(zcn.NetTotal, 2) AS NetTotal,
    zcn.InvoiceDate AS CreditNoteDate,
    CASE
        WHEN ca.SessionDate IS NULL THEN 'ORPHAN (no session date)'
        WHEN ca.SessionDate >= '2026-03-01 02:00:00' AND ca.SessionDate < '2026-03-02 02:00:00' THEN 'IN SYNC DATE'
        ELSE 'DIFFERENT SESSION DATE'
    END AS SyncStatus
FROM tblZatcaCreditNote zcn
LEFT JOIN tblCashierActivity ca ON zcn.SessionID = ca.SessionID
WHERE zcn.InvoiceDate >= '2026-03-01 02:00:00' AND zcn.InvoiceDate < '2026-03-02 02:00:00'
ORDER BY zcn.SessionID, zcn.InvoiceID;

Q10d: Compare CRA vouchers vs ZatcaCreditNote for a date
-----------------------------------------------------------
-- Shows how many credit notes are captured by CRA vs total issued
-- Uses 02:00-02:00 business day window
SELECT 'CRA vouchers (code uses these)' AS Source, COUNT(*) AS Cnt,
    ROUND(SUM(cra.ReturnAmount), 2) AS TotalAmount
FROM tblCashierActivityReturnAmount cra
WHERE cra.ReturnDate >= '2026-03-01 02:00:00' AND cra.ReturnDate < '2026-03-02 02:00:00'

UNION ALL

SELECT 'ZatcaCreditNote (all issued)', COUNT(*),
    ROUND(SUM(zcn.NetTotal), 2)
FROM tblZatcaCreditNote zcn
WHERE zcn.InvoiceDate >= '2026-03-01 02:00:00' AND zcn.InvoiceDate < '2026-03-02 02:00:00'

UNION ALL

SELECT 'ZatcaCreditNote in synced sessions only', COUNT(*),
    ROUND(SUM(zcn.NetTotal), 2)
FROM tblZatcaCreditNote zcn
INNER JOIN tblCashierActivity ca ON zcn.SessionID = ca.SessionID
WHERE zcn.InvoiceDate >= '2026-03-01 02:00:00' AND zcn.InvoiceDate < '2026-03-02 02:00:00'
  AND ca.SessionDate >= '2026-03-01 02:00:00' AND ca.SessionDate < '2026-03-02 02:00:00';

Q16: Vendors with purchase invoices

Q19: Overall database counts
-------------------------------
SELECT 'tblInvoice total' AS metric, COUNT(*) AS cnt FROM tblInvoice
UNION ALL SELECT 'tblInvoice IsReturned=1', COUNT(*) FROM tblInvoice WHERE IsReturned=1
UNION ALL SELECT 'tblPurchaseInvoice total', COUNT(*) FROM tblPurchaseInvoice
UNION ALL SELECT 'tblPurchaseInvoice IsReturn=0 Posted=1', COUNT(*) FROM tblPurchaseInvoice WHERE IsReturn=0 AND Posted=1
UNION ALL SELECT 'tblPurchaseInvoice IsReturn=1', COUNT(*) FROM tblPurchaseInvoice WHERE IsReturn=1
UNION ALL SELECT 'tblPurchaseInvoice IsReturn=1 Posted=1', COUNT(*) FROM tblPurchaseInvoice WHERE IsReturn=1 AND Posted=1
UNION ALL SELECT 'tblCashierActivityReturnAmount', COUNT(*) FROM tblCashierActivityReturnAmount
UNION ALL SELECT 'tblInvoiceReturnCode', COUNT(*) FROM tblInvoiceReturnCode
UNION ALL SELECT 'tblZatcaCreditNote', COUNT(*) FROM tblZatcaCreditNote
UNION ALL SELECT 'tblZatcaCreditNoteDetail', COUNT(*) FROM tblZatcaCreditNoteDetail
UNION ALL SELECT 'PT5 sessions (PaymentType=5)', COUNT(*) FROM tblCashierActivityDetail WHERE PaymentType=5
UNION ALL SELECT 'tblItems', COUNT(*) FROM tblItems
UNION ALL SELECT 'tblSuppliers', COUNT(*) FROM tblSuppliers
UNION ALL SELECT 'tblCustomers', COUNT(*) FROM tblCustomers
UNION ALL SELECT 'tblCashierActivity (sessions)', COUNT(*) FROM tblCashierActivity
UNION ALL SELECT 'tblSuppliersPayment', COUNT(*) FROM tblSuppliersPayment;
